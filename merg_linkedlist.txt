# merg two linkedlist
def merge_linked_list(h1, h2):
    dummy = Node(0)      # dummy start node
    current = dummy

    while h1 and h2:
        if h1.data < h2.data:
            current.next = h1    # h1 attach karo
            h1 = h1.next
        else:
            current.next = h2    # h2 attach karo
            h2 = h2.next
        current = current.next

    # jo bacha usey attach karo
    if h1:
        current.next = h1
    if h2:
        current.next = h2

    return dummy.next    # dummy ke baad se result shuru
