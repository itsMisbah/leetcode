# most repeative char in a string
def repeative_char(s):
    seen = {}
    
    for char in s:
        if char in seen:
            seen[char] += 1
        else:
            seen[char] = 1
            
    max_count = 0
    max_char = None
    
    keys = list(seen.keys())
    for key in keys:
        if seen[key] > max_count:
            max_count = seen[key]
            max_char = key
            
    return max_char
    
        
print(repeative_char("abcdddfiiifffffg"))

#time complexity O(n)
